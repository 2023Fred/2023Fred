Skill 1 — Criar CDB (DBCA Silent)

dbca -silent -createDatabase \
  -templateName General_Purpose.dbc \
  -gdbName CDBLAB -sid CDBLAB \
  -createAsContainerDatabase true \
  -numberOfPDBs 0 \
  -characterSet AL32UTF8 \
  -memoryMgmtType AUTO_SGA -totalMemory 2048 \
  -storageType FS \
  -datafileDestination /u01/app/oracle/oradata \
  -recoveryAreaDestination /u01/app/oracle/fast_recovery_area \
  -recoveryAreaSize 10240 \
  -enableArchive true \
  -emConfiguration NONE

Skill 2 — Criar PDB a partir do SEED

CREATE PLUGGABLE DATABASE pdb_app1
  ADMIN USER pdbadmin IDENTIFIED BY Oracle_4U
  ROLES = (DBA)
  FILE_NAME_CONVERT = (
    '/u01/app/oracle/oradata/CDBLAB/pdbseed/',
    '/u01/app/oracle/oradata/CDBLAB/pdb_app1/'
  )
  STORAGE (MAXSIZE 20G)
  DEFAULT TABLESPACE users
    DATAFILE '/u01/app/oracle/oradata/CDBLAB/pdb_app1/users01.dbf'
    SIZE 500M AUTOEXTEND ON MAXSIZE 8G;

ALTER PLUGGABLE DATABASE pdb_app1 OPEN;
ALTER PLUGGABLE DATABASE pdb_app1 SAVE STATE;

Skill 3 — Unplug/Replug (clone no mesmo CDB)

-- 1. Fechar e gerar manifest
ALTER PLUGGABLE DATABASE pdb_app1 CLOSE IMMEDIATE;
ALTER PLUGGABLE DATABASE pdb_app1 UNPLUG INTO '/u01/transfer/pdb_app1.xml';

-- 2. Remover do catálogo mantendo arquivos
DROP PLUGGABLE DATABASE pdb_app1 KEEP DATAFILES;

-- 3. Replicar (NOCOPY = zero cópia de dados)
CREATE PLUGGABLE DATABASE pdb_app1
  USING '/u01/transfer/pdb_app1.xml'
  NOCOPY TEMPFILE REUSE;
ALTER PLUGGABLE DATABASE pdb_app1 OPEN;
-- Resultado: GUID preservado · CON_ID novo (3→4) · <5 segundos

Skill 5 — Migração non-CDB → PDB

-- 1. Source em READ ONLY
SHUTDOWN IMMEDIATE; STARTUP MOUNT;
ALTER DATABASE OPEN READ ONLY;

-- 2. Gerar manifest
BEGIN DBMS_PDB.DESCRIBE(pdb_descr_file => '/u01/transfer/noncdb_manifest.xml'); END;
/

-- 3. Plugar no CDB destino
CREATE PLUGGABLE DATABASE pdb_noncdb
  USING '/u01/transfer/noncdb_manifest.xml'
  NOCOPY TEMPFILE REUSE;

-- 4. CRÍTICO: converter dicionário (~15 min)
ALTER SESSION SET CONTAINER = pdb_noncdb;

-- 5. Abrir

ALTER PLUGGABLE DATABASE pdb_noncdb SAVE STATE;

Skill 6 — Refreshable PDB (arquitetura)

-- Usuário comum para o clone (no source)
CREATE USER c##refresh_user IDENTIFIED BY Oracle_4U
  DEFAULT TABLESPACE users CONTAINER = ALL;
GRANT CREATE SESSION, CREATE PLUGGABLE DATABASE, SYSOPER
  TO c##refresh_user CONTAINER = ALL;

-- Refreshable PDB com refresh automático a cada 60 min
CREATE PLUGGABLE DATABASE pdb_app1_dev FROM pdb_app1@lk_prod
  FILE_NAME_CONVERT = (...)
  REFRESH MODE EVERY 60 MINUTES;

-- Refresh manual
ALTER PLUGGABLE DATABASE pdb_app1_dev REFRESH;

-- Converter para regular (irreversível)
ALTER PLUGGABLE DATABASE pdb_app1_dev REFRESH MODE NONE;
ALTER PLUGGABLE DATABASE pdb_app1_dev OPEN;

Skill 7 — Application Container (ERP)

-- Instalação da aplicação
ALTER SESSION SET CONTAINER = app_erp_root;
ALTER PLUGGABLE DATABASE APPLICATION erp_app BEGIN INSTALL '1.0';
-- ... criação de schema (tabelas CUSTOMERS, ORDERS, PRODUCT_CATALOG) ...
ALTER PLUGGABLE DATABASE APPLICATION erp_app END INSTALL '1.0';

-- Upgrade em cascata: v1.1 → adiciona schema completo
ALTER PLUGGABLE DATABASE APPLICATION erp_app BEGIN UPGRADE '1.0' TO '1.1';
ALTER PLUGGABLE DATABASE APPLICATION erp_app END UPGRADE TO '1.1';

-- v1.2 → coluna PHONE + Produto D
ALTER PLUGGABLE DATABASE APPLICATION erp_app BEGIN UPGRADE '1.1' TO '1.2';
ALTER TABLE erp_owner.customers ADD (phone VARCHAR2(20));
INSERT INTO erp_owner.product_catalog VALUES (4,'Produto D',199.90,'Y');
ALTER PLUGGABLE DATABASE APPLICATION erp_app END UPGRADE TO '1.2';

-- Sincronizar tenants
ALTER SESSION SET CONTAINER = app_pdb_cliente_a;
ALTER PLUGGABLE DATABASE APPLICATION erp_app SYNC;
ALTER SESSION SET CONTAINER = app_pdb_cliente_b;
ALTER PLUGGABLE DATABASE APPLICATION erp_app SYNC;

ALTER SESSION SET CONTAINER = pdb_app1;
DROP TABLE pdbadmin.vendas_teste;  -- ACIDENTE!

-- Flashback ao restore point

Skill 9 — Usuários Common e Local

-- Common user (em CDB$ROOT)
ALTER SESSION SET CONTAINER = CDB$ROOT;
CREATE USER c##dba_admin IDENTIFIED BY Oracle_4U
  DEFAULT TABLESPACE users TEMPORARY TABLESPACE temp
  CONTAINER = ALL;
GRANT DBA TO c##dba_admin CONTAINER = ALL;

-- CONTAINER_DATA para visão cross-PDB
ALTER USER c##monitor SET CONTAINER_DATA = ALL CONTAINER = CURRENT;

-- Local user (dentro do PDB)
ALTER SESSION SET CONTAINER = pdb_app1;
CREATE USER app_owner IDENTIFIED BY Oracle_4U
  DEFAULT TABLESPACE users QUOTA UNLIMITED ON users;
GRANT CREATE SESSION, CREATE TABLE, CREATE VIEW,
      CREATE SEQUENCE, CREATE PROCEDURE TO app_owner;

-- Validação de isolamento
-- APP_OWNER em PDB_NONCDB → no rows (não vaza entre PDBs)
-- APP_OWNER em PDB_APP1   → EXISTS (isolamento perfeito)
-- C##MONITOR cross-PDB    → 8 PDBs visíveis em uma query
